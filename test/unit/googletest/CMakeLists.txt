include(CheckLanguage)

if (CMAKE_CXX_COMPILER)
  configure_file(valgrind.suppressions valgrind.suppressions COPYONLY)

  include_directories("${DEP_INSTALL_PREFIX}/googletest/include")
  set(TEST_SRC
    OutputBuffers.cpp
    PoolArgs.cpp
    QueuePerThreadPool.cpp
    SinglyLinkedList.cpp
    addqueryfuncs.cpp
    bf.cpp
    compress.cpp
    debug.cpp
    handle_sql.cpp
    histogram.cpp
    print.cpp
    query_user_str.cpp
    str.cpp
    template_db.cpp
    trace.cpp
    trie.cpp
    xattrs.cpp
  )

  if (QPTPOOL_SWAP)
    list(APPEND TEST_SRC
      swap.cpp
    )
  endif()

  # these test files use CMake variables
  set(NEED_PATH
    BottomUp.cpp
    aggregate.cpp
    dbutils.cpp
    descend.cpp
    external.cpp
    utils.cpp
  )
  foreach(SRC ${NEED_PATH})
    configure_file("${SRC}.in" "${SRC}" @ONLY)
    list(APPEND TEST_SRC
      "${CMAKE_CURRENT_BINARY_DIR}/${SRC}"
    )
  endforeach()

  add_executable(unit_tests main.cpp ${TEST_SRC} $<TARGET_OBJECTS:gufi_query_lib>)
  target_link_libraries(unit_tests -L"${DEP_INSTALL_PREFIX}/googletest/lib" -L"${DEP_INSTALL_PREFIX}/googletest/lib64" gtest ${COMMON_LIBRARIES})

  add_test(NAME unit_tests COMMAND "${CMAKE_CURRENT_BINARY_DIR}/unit_tests" --gtest_shuffle
    WORKING_DIRECTORY "${TEST_WORKING_DIRECTORY}")
  set_tests_properties(unit_tests PROPERTIES LABELS "unit")
endif()
