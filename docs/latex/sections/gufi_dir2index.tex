\subsubsection{\gufidirindex}
\gufidirindex is used to directly create an index based off of the
contents of a provided directory.

\begin{table} [h!]
  \centering
  \begin{tabularx}{\textwidth}{|l|X|}
    \hline
    Flag & Functionality \\
    \hline
    -h, -{}-help & help \\
    \hline
    -H, -{}-debug & show assigned input values (debugging) \\
    \hline
    -v, -{}-version & version \\
    \hline
    -n, -{}-threads \textless n\textgreater & number of threads \\
    \hline
    -{}-min-level \textless n\textgreater & minimum level to descend to \\
    \hline
    -{}-max-level \textless n\textgreater & maximum level to descend to \\
    \hline
    -{}-path-list \textless filename\textgreater & File containing paths at
    single level to walk (not including starting path). If -{}-min-level
    \textgreater \ 0, prepend each line of the file with the index path. \\
    \hline
    -x, -{}-index-xattrs & index xattrs \\
    \hline
    -q & check that external databases are valid before tracking during indexing \\
    \hline
    -{}-plugin \textless library\_name\textgreater & plugin library for modifying db entries \\
    \hline
    -{}-target-memory \textless bytes\textgreater & target memory utilization (soft limit) \\
    \hline
    -{}-swap-prefix \textless path\textgreater & File name prefix for swap files \\
    \hline
    -{}-subdir-limit \textless count\textgreater & Number of subdirectories allowed to be enqueued for parallel processing. Any remainders will be processed serially \\
    \hline
    -{}-compress & compress work items \\
    \hline
  \end{tabularx}
  \caption{\gufidirindex Flags and Arguments}
\end{table}

\begin{figure} [h!]
  \centering
  \includegraphics[width=1.0\textwidth]{images/gufi_dir2index.png}
  \caption{\gufidirindex workflow}
\end{figure}

\paragraph{Usage} ~\\\\
\gufidirindex \texttt{[flags] dir... GUFU\_tree\_parent} \\\\
The index of \texttt{dir\textsubscript{i}} will be placed in \texttt{GUFI\_tree\_parent/\$(basename dir\textsubscript{i})}.
